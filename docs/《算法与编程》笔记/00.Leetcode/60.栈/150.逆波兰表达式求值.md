> * 遇到数字则入栈；遇到算符则取出栈顶两个数字进行计算，并将结果压入栈中。
```js
function evalRPN(tokens: string[]): number {
    let stack: number[] = []
    let temp: number
    for(let i = 0; i < tokens.length; i++) {
        switch (tokens[i]) {
            case '+' :
                temp = stack.pop()!
                temp = temp + stack.pop()!
                stack.push(temp)
                break
            case '-' :
                temp = stack.pop()!
                temp = stack.pop()! - temp  //注意运算顺序
                stack.push(temp)
                break
            case '*' :
                temp = stack.pop()!
                temp = stack.pop()! * temp
                stack.push(temp)
                break
            case '/' :
                temp = stack.pop()!
                temp = Math.trunc(stack.pop()! / temp)
                stack.push(temp)
                break
            default :
                stack.push(Number(tokens[i]))
                break
        }
    }
    return stack.pop()!
};

```
