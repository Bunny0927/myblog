```js
/**
 * @param {number} target
 * @param {number[]} nums
 * @return {number}
 */
var minSubArrayLen = function(target, nums) {
    let l = 0, r = 0, sum = 0;
    let ans = nums.length + 1;
    for (let r = 0; r < nums.length; r++) {
        sum += nums[r];
        while (sum >= target) {
            ans = Math.min(ans, r - l + 1);
            sum -= nums[l];
            l += 1;
        }
    }
    if (ans == nums.length + 1) {
        return 0
    } else {
        return ans;
    }
};
```
