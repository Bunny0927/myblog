```js
var findPeakElement = function (nums) {
    var l = 0;
    var r = nums.length - 1;
    while (l < r) {
        var mid = Math.floor(l + (r - l) / 2);
        if (nums[mid] < nums[mid + 1]) {
            l = mid + 1;
        }
        else if (nums[mid] > nums[mid + 1]) {
            r = mid;
        }
    }
    return l;
};
```
