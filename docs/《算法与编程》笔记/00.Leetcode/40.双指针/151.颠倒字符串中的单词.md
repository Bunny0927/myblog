function reverseWords(s: string): string {
    // 去除字符串中多余的空格
    function removeSpeaces(arr: string[]) {
        let slow: number = 0, fast: number = 0
        while(fast < arr.length) {
            // 去除首部空格
            if(arr[fast] === ' ' && fast === 0) {
                fast++
            } else if (arr[fast] === ' ' && arr[fast - 1] === ' ') {
                fast++   //去除中间空格
            } else {
                arr[slow] = arr[fast]
                slow++
                fast++
            }
        }
        // 去除尾部剩下的一个空格
        if(arr[slow - 1] === ' ') {
            arr.length = slow - 1
        } else {
            arr.length = slow
        }
    }

    // 反转字符串
    function reverseStr(s: string[], start: number, end: number){
        let temp: string
        while(start < end) {
            temp = s[end]
            s[end] = s[start]
            s[start] = temp
            start++
            end--
        }
    }

    let strArr: string[] = s.split('')
    // 去除空格
    removeSpeaces(strArr)
    // 反转整个字符串
    reverseStr(strArr, 0, strArr.length - 1)
    // 反转整个单词
    let k: number = 0
    for(let i = 0; i <= strArr.length; i++) {
        if(strArr[i] === ' ' || i === strArr.length) {
            reverseStr(strArr, k, i - 1)
            k = i + 1
        }
    }
    return strArr.join('')
 };
