/**
 * @param {string} s
 * @return {boolean}
 */
var isPalindrome = function(s) {
    s= s.replace(/[^A-Za-z0-9]/g, ''); //使用正则表达式替换无效字符
    const s_lower = s.toLowerCase();
    let l = 0, r = s_lower.length - 1;
    while (l < r) {
        if (s_lower[l] !== s_lower[r]) {
            return false;
        } else {
            l++;
            r--;
        }
    }
    return true;
};
