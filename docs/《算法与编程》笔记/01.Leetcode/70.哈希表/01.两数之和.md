**使用哈希表减少运算复杂度**
```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
    hashMap = new Map();
    for(let index = 0; index < nums.length; index++) {
        if(hashMap.has(target-nums[index])) {
            return [hashMap.get(target-nums[index]), index]
        }
        hashMap.set(nums[index], index)
    }
    return []
};
```
