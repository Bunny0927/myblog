/**
 * @param {number[]} num
 * @param {number} k
 * @return {number[]}
 */
var addToArrayForm = function(num, k) {
    let l1 = num.length - 1;
    const result = [];
    let carry = 0;  //进位
    while (l1 >= 0 || k != 0) {
        const x = l1 >= 0 ? num[l1] : 0;
        const y = k != 0 ? k % 10 : 0;
        const sum = x + y + carry;
        result.push(sum % 10);
        carry = Math.floor(sum / 10);
        l1--;
        k = Math.floor(k / 10)
    }
    if(carry) result.push(carry);  //循环结束后如果仍存在进位，放入数组开头
    return result.reverse();
};
