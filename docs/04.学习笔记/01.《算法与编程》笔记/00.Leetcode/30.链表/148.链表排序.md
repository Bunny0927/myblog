/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var sortList = function(head) {
    const arr = [];
    while (head !== null) {
        arr.push(head.val);
        head = head.next;
    }
    arr.sort((a, b) => a - b);

    //先转为数组排序再转换为链表，效率不高
    let p2 = new ListNode(0)
    let p3 = p2
    arr.forEach(item => {
        p3.next = new ListNode(item)
        p3 = p3.next
    })
    return p2.next
};
