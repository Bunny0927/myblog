function groupAnagrams(strs: string[]): string[][] {
    let obj = new Map()
    let result: string[][] = []
    for(let i of strs) {
        let temp: string = i.split('').sort().join('') //重点
        if(!obj.has(temp)) {
            obj.set(temp, [i])
        } else {
            let arr = obj.get(temp)
            arr.push(i)
            obj.set(temp, arr)
        }
    }
    for(let [key, value] of obj) {
        result.push(value)
    }
    return result
};
